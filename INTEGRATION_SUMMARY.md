## üé® Example Component Pattern

```javascript
import React, { useState, useEffect } from 'react';
import { workerService } from './services';
import { getErrorMessage, formatRatingStars } from './utils';

function WorkerList() {
  const [workers, setWorkers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState('');

  useEffect(() => {
    loadWorkers();
  }, []);

  const loadWorkers = async () => {
    try {
      setLoading(true);
      const result = await workerService.searchWorkers({
        is_available: true,
        sort: 'rating',
        order: 'DESC'
      });
      setWorkers(result.data);
      setError('');
    } catch (err) {
      setError(getErrorMessage(err));
    } finally {
      setLoading(false);
    }
  };

  if (loading) return <div>ÿ¨ÿßÿ±Ÿä ÿßŸÑÿ™ÿ≠ŸÖŸäŸÑ...</div>;
  if (error) return <div style={{ color: 'red' }}>{error}</div>;

  return (
    <div>
      {workers.map(worker => (
        <div key={worker.id}>
          <h3>{worker.user.name}</h3>
          <p>{formatRatingStars(worker.average_rating, true)}</p>
        </div>
      ))}
    </div>
  );
}
```
